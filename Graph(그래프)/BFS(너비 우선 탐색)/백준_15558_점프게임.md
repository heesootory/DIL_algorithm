import java.io.*;
import java.util.*;

class Pair{
    int x;
    int y;
    public Pair(int x, int y){
        this.x = x;
        this.y = y;
    }
}

public class Main {
    static int n;
    static int k;
    static int[][] road;
    static boolean[][] visited;

    static boolean bfs(Pair p1){

        q.add(p1);

    public static void main(String[] args) throws IOException {
