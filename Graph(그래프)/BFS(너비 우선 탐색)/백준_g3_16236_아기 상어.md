class Shark{
    int x, y;

    Shark(int x, int y, int cnt, int size){
        this.x = x;
        this.y = y;
        this.cnt = cnt;
        this.size = size;
        grow();
    }
    void grow(){
        if(cnt == size){
            cnt = 0;
            size++;
        }
    }
}
